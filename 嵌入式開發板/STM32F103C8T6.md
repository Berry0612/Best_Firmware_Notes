```c
// 1. 定義暫存器 (一定要算對，差 1 bit 都不行)
// GPIOC Base: 0x4001 1000
#define RCC_APB2ENR  *(volatile unsigned int *)(0x40021018)
#define GPIOC_CRH    *(volatile unsigned int *)(0x40011004) // Base + 0x04
#define GPIOC_ODR    *(volatile unsigned int *)(0x4001100C) // Base + 0x0C

void main() {
    // -----------------------------------------
    // 初始化 (Initialization) - 只做一次
    // -----------------------------------------

    // 1. 開啟 GPIOC 時鐘 (Set Bit 4)
    RCC_APB2ENR |= (1 << 4);

    // 2. 設定 PC13 模式
    // 步驟 A: 清空 (Clear) PC13 的 4 個設定位元 (Bit 20-23)
    // 0xF 是 1111，移到位置 20，取反向變成遮罩，用 AND 清除
    GPIOC_CRH &= ~(0xF << 20); 

    // 步驟 B: 寫入 (Set) 新的模式
    // 目標: Output Push-Pull 50MHz -> 0011 (Binary) -> 0x3 (Hex)
    // 把 3 移到位置 20，用 OR 寫入
    GPIOC_CRH |= (3 << 20);


    // -----------------------------------------
    // 主迴圈 (Main Loop) - 無限執行
    // -----------------------------------------
    while(1) {
        // 3. LED 亮 (PC13 Output Low -> Clear Bit)
        // 你的寫法是對的，但記得位元運算優先級，建議加括號習慣一下
        GPIOC_ODR &= ~(1 << 13);
        
        // 4. 延遲 (隨便跑個迴圈耗時間)
        for(int i = 0; i < 500000; i++);

        // 5. LED 滅 (PC13 Output High -> Set Bit)
        GPIOC_ODR |= (1 << 13);

        // 6. 延遲
        for(int i = 0; i < 500000; i++);
    }
}
```
